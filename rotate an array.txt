
public class hi {

	public static void main(String args[])
	{
		
		int a[] = {1,2,3,4,5,6,7,8,9};
		rotate(a);
		
		for(int i=0;i<a.length;i++) {
			System.out.print(a[i]+" ");
		}
		
		
		
		
		
	}
	
	static void rotate(int a[]){
		//int a[] = {1,2,3,4,5,6,7,8,9};
		int temp,k=3,i;
		int n=a.length;
		
		while(k>0)
		{
			temp=a[n-1];
			for(i=n-2;i>=0;i--)
			{
				a[i+1]=a[i];
			}
			a[0]=temp;
			k--;
		}
		
		
	}
}
