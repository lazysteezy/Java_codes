class hi{
	public static void main(String args[])
	{
		int arr[] = {1,2,3,4,5,6,7,8};
		int c[] = new int[arr.length];
		int start=0;
		int end=arr.length-1;
		int k=0,j=0,i;
		
		while(start<=end )
		{
			if(start<=end && k==0)
			{
				c[j]=arr[end];
				j++;
				end--;
				k=1;
				
			}else if(start <= end && k==1)
			{
				c[j] = arr[start];
				j++;
				start++;
				k=0;
				
			}
		}
		for(i=0;i<arr.length;i++)
		{
			System.out.print(c[i] + " ");
		}
		
	}
}
