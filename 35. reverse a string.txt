class lazy
{ 
	public static void main(String args[])
	{
		
		String sc = "hello arun";
		char y[] = sc.toCharArray();
		int len=y.length;
		int start=0;
		int end=len-1;
		
		
		while(start <= end)
		{
			char temp = y[start];
			y[start] = y[end];
			y[end] =temp;
			start++;
			end--;
		}
		for(int i=0;i<y.length;i++)
		System.out.print(y[i]);
		
		}
		
}
