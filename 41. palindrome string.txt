import java.util.*;

class lazy
{ 
	public static void main(String args[])
	{
		Scanner sc = new Scanner(System.in);
		
		String sc1 = sc.nextLine();
		char x[] = sc1.toCharArray();
		
		int len=x.length;
		int start=0;
		int end=len-1;
		int temp=0;
		while(start < end)
		{
			if(x[start] != x[end])
			{
				temp=1;
			}
			start++;
			end--;
		}
		if(temp==0)
		{
			System.out.print("String is palindrome");
		}
		else
		{
			System.out.print("String isn't palindrome");
		}
		
}
}
