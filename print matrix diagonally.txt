#include <stdio.h>
#include <string.h>

int main(){


  int m,n;
  int i,j,k;
  int mat[50][50];


  printf("Enter the number of rows: \n");
  scanf("%d",&m);

  printf("Enter the number of column\n");
  scanf("%d",&n);

  printf("Enter the %d element in matrix: \n",m*n);
  for(i=0;i<m;i++)
  {
      for(j=0;j<n;j++)
      {
          scanf("%d",&mat[i][j]);
      }
  }



  printf("Matrix is:\n");

  for(i=0;i<m;i++)
  {
      for(j=0;j<n;j++)
      {
          printf("%d\t",mat[i][j]);
      }
      printf("\n");
  }

  printf("spiral form is: \n");

  for(k=0;k<=m-1;k++)
  {
      i=k;
      j=0;
      while(i>=0)
      {
          printf("%d\t",mat[i][j]);
          i--;j++;
      }
      printf("\n");
  }
  for(k=1;k<=n-1;k++)
  {
      i=m-1;
      j=k;
      while(j<=n-1)
      {
          printf("%d\t",mat[i][j]);
          i--;
          j++;
      }
      printf("\n");
  }



    return 0;
}
