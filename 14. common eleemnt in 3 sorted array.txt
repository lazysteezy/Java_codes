import java.util.*;

class lazy
{ 
	public static void main(String args[])
	{
		
		int a[] = {1,2,3,4,5,6,7};
		int b[] = {2,3,4,9};
		int c[] = {2,3};
		int x,y,z;
		x=y=z=0;
		
		while(x<a.length && y<b.length && z<c.length)
		{
			if(a[x]==b[y] && b[y]==c[z])
			{
				System.out.println(a[x]);
				x++;y++;z++;
			}
			else if(a[x]>b[y])
			{
				y++;
			}
			else if(b[y] > c[z])
			{
				z++;
			}
			else
			{
				x++;
			}
		}		
	}
	
}
