class hi{
	

	public static void main(String args[])
	{
		int a[] = {-2,-3,4,-1,-2,1,5,-3};
		int max_far = 0;
		int start,end,s;
		start=end=s=0;
		int i,max_end=0;
		
		for(i=0;i<a.length;i++)
		{
			max_end=max_end+a[i];
			if(max_far < max_end)
			{
				max_far=max_end;
				start=s;
				end=i;
			}
			if(max_end < 0)
			{
				max_end=0;
				s=i+1;
			}
		}
		System.out.print("total sum is "+ max_far + "\n");
		for(i=start;i<=end;i++)
		{
			System.out.print(a[i] + " ");
		}
		
		
		
		
}
}
